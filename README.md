# Currency_converter

## 1.  table名稱: **currencyconverter**


    create table currencyconverter
    (
    id         int auto_increment
        primary key,
    date       varchar(256) null,
    usd_to_ntd varchar(256) null,
    rmb_to_ntd varchar(256) null,
    usd_to_rmb varchar(256) null
    );
